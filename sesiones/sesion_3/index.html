Condicionales: if, else

          
            import random
            x = random.random()
            if x < 0.2:
              print "x es menor que 0.2"
            elif x <= 0.5:
              print "x es mayor que 0.2 y menor o igual que 0.5"
            else:
              print "x es mayor que 0.5"
